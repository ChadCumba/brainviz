canvas = document.getElementById("canvas");
context = canvas.getContext("2d");

var bmd = new BitmapData(400, 400);
bmd.perlinNoise(
	bmd.width/2,	// baseX
	bmd.height/2,	// baseY
	parseInt(Math.random()*0xffff),	// seed
	BitmapDataChannel.RED | BitmapDataChannel.BLUE,	// channel options
	false // gray-scale
);

context.putImageData(bmd.data, 0, 0);
