HTML5 Canvas API implementation of the AS3 BitmapData class.

canvas = document.getElementById("canvas");
context = canvas.getContext("2d");

bmd = new BitmapData(256, 256, false, 0xff0000);

color = 0x000000;
for(y=0; y<bmd.height; y++) {
	for(x=0; x<bmd.width; x++) {
		bmd.setPixel(x, y, color+=0xff);
	}
}

bmd.fillRect(new Rectangle(156, 156, 100, 100), 0xdd00dd);

context.putImageData(bmd.data, 0, 0);
