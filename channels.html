canvas = document.getElementById("canvas");
context = canvas.getContext("2d");

bikes = new Image();
bikes.src = "res/bikes.png";
bikesData = new BitmapData(300, 300);
bikesData.draw(bikes);

lenna = new Image();
lenna.src = "res/Lenna.png";
leenaData = new BitmapData(512, 512);
leenaData.draw(lenna);

bikesData.copyChannel(leenaData, new Rectangle(150, 150, 200, 200), new Point(50, 50), BitmapDataChannel.RED, BitmapDataChannel.GREEN);

context.putImageData(bikesData.data, 0, 0);
