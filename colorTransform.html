HTML5 Canvas API implementation of the AS3 BitmapData class.

canvas = document.getElementById("canvas");
context = canvas.getContext("2d");

bikes = new Image();
bikes.src = "res/bikes.png";
bikesData = new BitmapData(300, 300);
bikesData.draw(bikes);

colorTrans=new ColorTransform(1, 0, 1, 1, 180);
rect=new Rectangle(20, 20, 100, 100);

bikesData.colorTransform(rect, colorTrans);

context.putImageData(bikesData.data, 0, 0);
